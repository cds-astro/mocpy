Let’s create a MOC from the coordinates of this table

[3]:

moc = MOC.from_lonlat(
    table["_RAJ2000"].T * u.deg,
    table["_DEJ2000"].T * u.deg,
    max_norder=6,
)

[4]:

fig = plt.figure(figsize=(9, 5))

galactic_wcs = WCS(
    {
        "naxis": 2,
        "naxis1": 1620,
        "naxis2": 810,
        "crpix1": 810.5,
        "crpix2": 405.5,
        "cdelt1": -0.2,
        "cdelt2": 0.2,
        "ctype1": "GLON-AIT",
        "ctype2": "GLAT-AIT",
    },
)

ax = fig.add_subplot(1, 1, 1, projection=galactic_wcs, frame_class=EllipticalFrame)

moc.fill(
    ax=ax,
    wcs=galactic_wcs,
    edgecolor="teal",
    facecolor="orange",
    linewidth=1.0,
    fill=True,
    alpha=0.8,
)
moc.border(ax=ax, wcs=galactic_wcs, color="k")

ax.grid(visible=True)
